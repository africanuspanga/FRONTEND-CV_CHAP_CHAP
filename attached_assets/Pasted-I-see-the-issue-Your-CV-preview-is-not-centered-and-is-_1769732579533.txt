## The Problem

The CV preview container is not properly centered and the CV is overflowing/getting cut off.

## The Fix

Update your **Finalize Resume / Preview page** with this code:

**`src/app/(builder)/preview/page.tsx`**

```tsx
'use client';

import { useRouter } from 'next/navigation';
import { useCVStore } from '@/stores/cv-store';
import { TemplatePreview } from '@/components/templates/preview';
import { Button } from '@/components/ui/button';

export default function PreviewPage() {
  const router = useRouter();
  const { cvData, templateId, selectedColor } = useCVStore();

  return (
    <div className="min-h-screen bg-gray-100 flex flex-col">
      {/* Header */}
      <header className="bg-white border-b sticky top-0 z-20">
        <div className="max-w-7xl mx-auto px-4 py-4">
          {/* Back Button */}
          <button 
            onClick={() => router.back()}
            className="mb-4 text-gray-600 hover:text-gray-900 flex items-center gap-1"
          >
            ← Back
          </button>

          <h1 className="text-2xl md:text-3xl font-bold text-center text-gray-900 mb-4">
            Finalize Resume
          </h1>

          {/* Action Buttons */}
          <div className="flex justify-center gap-3">
            <Button
              variant="outline"
              onClick={() => router.push('/template?edit=true')}
              className="flex items-center gap-2"
            >
              ✏️ Change Template
            </Button>
            <Button
              variant="outline"
              onClick={() => router.push('/personal')}
              className="flex items-center gap-2"
            >
              ✏️ Edit Resume
            </Button>
          </div>
        </div>
      </header>

      {/* CV Preview - CENTERED */}
      <main className="flex-1 overflow-auto py-6 px-4">
        {/* This is the key: flex + justify-center + items-start */}
        <div className="flex justify-center items-start min-h-full">
          {/* 
            Preview Container:
            - overflow-x-auto allows horizontal scroll on small screens
            - The inner div scales the CV to fit
          */}
          <div className="overflow-x-auto">
            <div 
              className="bg-white shadow-2xl rounded-lg overflow-hidden mx-auto"
              style={{
                /* Scale down on mobile, full size on desktop */
                transform: 'scale(var(--preview-scale, 0.6))',
                transformOrigin: 'top center',
              }}
            >
              {/* 
                The actual CV at A4 dimensions
                210mm x 297mm = ~794px x 1123px at 96dpi
              */}
              <div style={{ width: '210mm', minHeight: '297mm' }}>
                <TemplatePreview
                  templateId={templateId}
                  data={cvData}
                  colorOverride={selectedColor}
                  scale={1}
                />
              </div>
            </div>
          </div>
        </div>
      </main>

      {/* Bottom CTA */}
      <footer className="sticky bottom-0 bg-white border-t p-4 shadow-lg">
        <div className="max-w-md mx-auto">
          <Button 
            onClick={() => router.push('/payment')}
            className="w-full h-14 text-lg bg-blue-600 hover:bg-blue-700"
          >
            Save & Next
          </Button>
        </div>
      </footer>

      {/* CSS Variable for responsive scaling */}
      <style jsx global>{`
        :root {
          --preview-scale: 0.45;
        }
        
        @media (min-width: 640px) {
          :root {
            --preview-scale: 0.55;
          }
        }
        
        @media (min-width: 768px) {
          :root {
            --preview-scale: 0.65;
          }
        }
        
        @media (min-width: 1024px) {
          :root {
            --preview-scale: 0.75;
          }
        }
        
        @media (min-width: 1280px) {
          :root {
            --preview-scale: 0.85;
          }
        }
      `}</style>
    </div>
  );
}
```

## Alternative: Simpler Fixed-Scale Version

If the CSS variables approach doesn't work well, use this simpler version:

```tsx
'use client';

import { useRouter } from 'next/navigation';
import { useState, useEffect } from 'react';
import { useCVStore } from '@/stores/cv-store';
import { TemplatePreview } from '@/components/templates/preview';
import { Button } from '@/components/ui/button';

export default function PreviewPage() {
  const router = useRouter();
  const { cvData, templateId, selectedColor } = useCVStore();
  const [scale, setScale] = useState(0.5);

  // Calculate scale based on screen width
  useEffect(() => {
    const calculateScale = () => {
      const screenWidth = window.innerWidth;
      const cvWidth = 794; // 210mm in pixels
      const padding = 32; // 16px each side
      const availableWidth = screenWidth - padding;
      
      // Calculate scale to fit CV in viewport
      let newScale = availableWidth / cvWidth;
      
      // Clamp between 0.35 and 0.9
      newScale = Math.max(0.35, Math.min(0.9, newScale));
      
      setScale(newScale);
    };

    calculateScale();
    window.addEventListener('resize', calculateScale);
    return () => window.removeEventListener('resize', calculateScale);
  }, []);

  return (
    <div className="min-h-screen bg-gray-100 flex flex-col">
      {/* Header */}
      <header className="bg-white border-b sticky top-0 z-20 px-4 py-4">
        <button 
          onClick={() => router.back()}
          className="absolute left-4 top-4 text-gray-600"
        >
          ←
        </button>

        <h1 className="text-2xl font-bold text-center text-gray-900 mb-4">
          Finalize Resume
        </h1>

        <div className="flex justify-center gap-3">
          <Button
            variant="outline"
            size="sm"
            onClick={() => router.push('/template?edit=true')}
          >
            ✏️ Change Template
          </Button>
          <Button
            variant="outline"
            size="sm"
            onClick={() => router.push('/personal')}
          >
            ✏️ Edit Resume
          </Button>
        </div>
      </header>

      {/* CV Preview - PERFECTLY CENTERED */}
      <main className="flex-1 overflow-auto py-6">
        <div 
          className="flex justify-center"
          style={{
            /* Account for scaled height to prevent extra scroll */
            minHeight: `${297 * scale}mm`,
          }}
        >
          <div
            className="bg-white shadow-2xl rounded-lg overflow-hidden"
            style={{
              transform: `scale(${scale})`,
              transformOrigin: 'top center',
              width: '210mm',
              minHeight: '297mm',
            }}
          >
            <TemplatePreview
              templateId={templateId}
              data={cvData}
              colorOverride={selectedColor}
              scale={1}
            />
          </div>
        </div>
      </main>

      {/* Bottom CTA */}
      <footer className="sticky bottom-0 bg-blue-600 p-4">
        <Button 
          onClick={() => router.push('/payment')}
          className="w-full h-14 text-lg bg-blue-600 hover:bg-blue-700 border-0"
        >
          Save & Next
        </Button>
      </footer>
    </div>
  );
}
```

---

## Key Fixes Applied

| Issue | Fix |
|-------|-----|
| CV not centered | `flex justify-center` on parent container |
| CV cut off on right | `transformOrigin: 'top center'` instead of `top left` |
| CV too big on mobile | Dynamic scale calculation based on screen width |
| Awkward scrolling | Scale CV to fit viewport width |
| Bottom button not sticky | `sticky bottom-0` with proper z-index |
